\textbf{Rotemberg and Woodford (1999): second-order approximation to household utility around efficient allocation}
\[
\tilde y_t=y_t-y_t^n,
\qquad
Y_t=A_tN_t^{1-\alpha}\Rightarrow \tilde y_t=\left(1-\alpha\right)\tilde n_t
\Rightarrow \tilde n_t=\frac{\tilde y_t}{1-\alpha}.
\]
\[
\text{expand utility around }x=\tilde y_t=\tilde c_t,\qquad
U=\frac{C^{1-\delta}}{1-\delta}-\frac{N^{1+\psi}}{1+\psi}.
\]
\purp{consumption moves by $x$, labor moves by $x/\left(1-\alpha\right)$.}
\par\vspace{1em}
\textbf{Consumption part:} \purp{move by $x$}
\[
\frac{\left(C^ne^x\right)^{1-\sigma}-\left(C^n\right)^{1-\sigma}}{\left(1-\sigma\right)U_c^nC^n}\tag{$\ast$}
\qquad\text{where }U_c^n=\left(C^n\right)^{-\sigma}
\]
\[
\left.
\begin{aligned}
\left(\ast\right)&=\frac{e^{\left(1-\sigma\right)x}-1}{1-\delta},\\[-1pt]
 e^{\left(1-\sigma\right)x}&\simeq1+\left(1-\sigma\right)x+\frac12\left(1-\sigma\right)^2x^2
\end{aligned}\right\}
\Rightarrow\quad
\left(\ast\right)=x+\frac12\left(1-\sigma\right)x^2.
\]
\purp{Taylor expansion around $0$.}
\par\vspace{1em}
\textbf{Labor part:} Since consumption move by $x$, labor move by $x/\left(1-\alpha\right)$.

Expand labor by $\dfrac{x}{1-\alpha}$ gives the normalized labor utility change:
\[
-\frac{\left(N^ne^{x/\left(1-\alpha\right)}\right)^{1+\psi}-\left(N^n\right)^{1+\psi}}{\left(1+\psi\right)U_c^nC^n}\tag{$\ast\ast$}
\]
\begin{align*}
\text{Since}\quad -\frac{U_N}{U_C}&=MPN=\left(1-\alpha\right)\frac{Y^n}{N^n},
& U_N&=\left(N^n\right)^\psi,\\[-1pt]
\Rightarrow\quad
\frac{\left(N^n\right)^\psi}{U_c^n}&=\left(1-\alpha\right)\frac{C^n}{N^n},
&\Rightarrow\quad
\frac{\left(N^n\right)^{1+\psi}}{U_c^nC^n}&=1-\alpha.
\end{align*}
\begin{align*}
\left(\ast\ast\right)&=-\frac{e^{\left(1+\psi\right)x/\left(1-\alpha\right)}-1}{1+\psi}
\frac{\left(N^n\right)^{1+\psi}}{U_c^nC^n}\\[-1pt]
&=-\left(1-\alpha\right)\frac{e^{\left(1+\psi\right)x/\left(1-\alpha\right)}-1}{1+\psi}.
\end{align*}
\[
e^{\left(1+\psi\right)x/\left(1-\alpha\right)}\simeq
1+\frac{1+\psi}{1-\alpha}x
+\frac12\left(\frac{1+\psi}{1-\alpha}\right)^2x^2,
\qquad\text{expand at }x=0.
\]
\[
\left(\ast\ast\right)=-\frac{1-\alpha}{1+\psi}
\left(\frac{1+\psi}{1-\alpha}x+\frac12\left(\frac{1+\psi}{1-\alpha}\right)^2x^2\right)
=-x-\frac12*\frac{1+\psi}{1-\alpha}x^2.
\]
\par\vspace{1em}
\textbf{Combine consumption part with labor part:}
\begin{align*}
&x+\frac12\left(1-\sigma\right)x^2-x-\frac12\frac{1+\psi}{1-\alpha}x^2\\[-1pt]
&\qquad=\frac12\left(1-\sigma-\frac{1+\psi}{1-\alpha}\right)x^2
=-\frac12\left(\sigma-1+\frac{1+\psi}{1-\alpha}\right)x^2
=-\frac12\left(\sigma+\frac{\alpha+\psi}{1-\alpha}\right)\tilde y_t^2,
\quad \left(x=\tilde y\right).
\end{align*}

\textbf{Now derive the inflation part:} $P_t\left(i\right)\neq P_t\left(j\right)$; inflation creates relative-price dispersion.
\[
\begin{aligned}
&\text{Cross sectional variance of log relative price}\Rightarrow
V=\int_0^1\left(p\left(i\right)-p\right)^2di=\int_0^1s\left(i\right)^2di,\\[-1pt]
&\hspace{4em} s\left(i\right)=p\left(i\right)-p.
\end{aligned}
\]
\begin{align*}
Y\left(i\right)&=\left(\frac{P\left(i\right)}{P}\right)^{-\varepsilon}Y
\Rightarrow \log Y\left(i\right)=-\varepsilon\log P\left(i\right)+\varepsilon\log P+\log Y,\\[-1pt]
\Rightarrow\quad y\left(i\right)-y&=-\varepsilon\left(p\left(i\right)-p\right)=-\varepsilon s\left(i\right).
\end{align*}
\[
\left.
\begin{aligned}
Y\left(i\right)&=A N\left(i\right)^{1-\alpha}
\Rightarrow
N\left(i\right)=\left(\frac{Y\left(i\right)}{A}\right)^{\frac{1}{1-\alpha}}\\
Y\left(i\right)&=Ye^{-\varepsilon s\left(i\right)}
\end{aligned}
\right\}
\Rightarrow
N\left(i\right)=\left(\frac{Y}{A}\right)^{\frac{1}{1-\alpha}}
e^{-\frac{\varepsilon}{1-\alpha}s\left(i\right)}.
\]
\[
N=\int_0^1N\left(i\right)di=\left(\frac{Y}{A}\right)^{\frac{1}{1-\alpha}}
\int_0^1e^{-\frac{\varepsilon}{1-\alpha}s\left(i\right)}di,
\qquad
\tag{1}
\text{ithout price dispersion }s\left(i\right)=0.
\]
\[
\Rightarrow
N^{sym}=\left(\frac{Y}{A}\right)^{\frac{1}{1-\alpha}}
\tag{2}
\quad\cdots\quad \text{symmetrical labor/price.}
\]
\[
\frac{\left(1\right)}{\left(2\right)}=\frac{N}{N^{sym}}=\int_0^1e^{-\frac{\varepsilon}{1-\alpha}s\left(i\right)}di,
\qquad
\text{let }a=\frac{\varepsilon}{1-\alpha}.
\]
\[
e^{-as\left(i\right)}\simeq1-as\left(i\right)+\frac{a^2}{2}s\left(i\right)^2,
\qquad\text{expand around }s\left(i\right)=0.
\]
\[
\Rightarrow\quad
\frac{N}{N^{sym}}=1-a\int_0^1s\left(i\right)di+\frac{a^2}{2}\int_0^1s\left(i\right)^2di.\tag{$\ast\ast\ast$}
\]
\begin{align*}
P
&=
\left[
\int_0^1 P\left(i\right)^{1-\varepsilon}\,di
\right]^{\frac{1}{1-\varepsilon}}
\\[2pt]
\Rightarrow\quad
1
&=
\left[
\int_0^1
\left(\frac{P\left(i\right)}{P}\right)^{1-\varepsilon}
di
\right]^{\frac{1}{1-\varepsilon}}
\\[2pt]
\Rightarrow\quad
1
&=
\int_0^1 e^{\left(1-\varepsilon\right)s\left(i\right)}\,di,
\qquad
\frac{P\left(i\right)}{P}
=
e^{p\left(i\right)-p}
=
e^{s\left(i\right)}.
\end{align*}

\[
e^{\left(1-\varepsilon\right)s\left(i\right)}
\approx
1+\left(1-\varepsilon\right)s\left(i\right)
+\frac12\left(1-\varepsilon\right)^2 s\left(i\right)^2.
\]

\begin{align*}
1
&=
\int_0^1
\left[
1+\left(1-\varepsilon\right)s\left(i\right)
+\frac12\left(1-\varepsilon\right)^2 s\left(i\right)^2
\right]di
\\[2pt]
&=
1+\left(1-\varepsilon\right)\int_0^1 s\left(i\right)\,di
+\frac12\left(1-\varepsilon\right)^2\int_0^1 s\left(i\right)^2\,di.
\end{align*}
\[
\Rightarrow\quad
\left(1-\varepsilon\right)\int_0^1 s\left(i\right)\,di
+\frac12\left(1-\varepsilon\right)^2 V=0,
\qquad
\text{where }
V \equiv \int_0^1 s\left(i\right)^2\,di .
\]

\[
\Rightarrow\quad
\int_0^1 s\left(i\right)\,di
+\frac12\left(1-\varepsilon\right)V=0
\quad\Rightarrow\quad
\int_0^1 s\left(i\right)\,di
=
\frac{\varepsilon-1}{2}V.
\]

\begin{equation}
\begin{aligned}
\left(\ast\ast\ast\right)
=\frac{N}{N^{sym}}
&=
1-a\int_0^1 s\left(i\right)\,di
+\frac{a^2}{2}\int_0^1 s\left(i\right)^2\,di
\\[2pt]
&=
1-a\cdot\frac{\varepsilon-1}{2}V
+\frac{a^2}{2}V
\\[2pt]
&=
1-\frac{\varepsilon}{1-\alpha}\cdot\frac{\varepsilon-1}{2}V
+\frac12\left(\frac{\varepsilon}{1-\alpha}\right)^2V
\\[2pt]
&=
1+
\frac{\varepsilon\left(1-\alpha+\alpha\varepsilon\right)}
{2\left(1-\alpha\right)^2}V.
\end{aligned}
\tag{$\ast\ast\ast\ast$}
\end{equation}

\green{Price dispersion increases labor required to produce the same aggregate output.}

\begin{align*}
-\frac{U_N}{U_C}&=MPN=\left(1-\alpha\right)\frac{Y}{N},
&\purp{\text{We need to know extra labor's utility loss.}}
\end{align*}
\[
\left.
\begin{aligned}
\frac{\MPN\,\Delta N}{Y}
&=\left(1-\alpha\right)\frac{\Delta N}{N},\\[4pt]
\left(\ast\ast\ast\ast\right)\quad
\frac{N-N^{sym}}{N^{sym}}
&=
\frac{N}{N^{sym}}-1
=
\frac{\varepsilon\left(1-\alpha+\alpha\varepsilon\right)}
{2\left(1-\alpha\right)^2}V
\end{aligned}
\right\}
\Rightarrow
\underbrace{
\frac{\MPN\,\Delta N}{Y}
}_{\text{labor price dispersion}}
=
\frac{\varepsilon\left(1-\alpha+\alpha\varepsilon\right)}
{2\left(1-\alpha\right)}V .
\]
\par\vspace{1em}
\[
\text{Define }\quad
\Theta=\frac{1-\alpha}{1-\alpha+\alpha\varepsilon}
\quad\text{(same as lecture 8)}
\quad\Rightarrow\quad
\frac{MPN\Delta N}{Y}=\frac12\frac{\varepsilon}{\Theta}V.
\]
Now convert price dispersion to inflation. Since
\[
P_t^{1-\varepsilon}
=\left(1-\theta\right)\left(P_t^*\right)^{1-\varepsilon}
+\theta P_{t-1}^{1-\varepsilon},
\]
log-linearizing around the zero-inflation steady state gives
\[
p_t=\left(1-\theta\right)p_t^*+\theta p_{t-1},
\qquad
p_t\equiv \log P_t .
\]
\begin{align*}
&\Rightarrow\quad
\pi_t=p_t-p_{t-1}
=\left(1-\theta\right)\left(p_t^*-p_{t-1}\right) \\[-1pt]
&\hspace{4.1em}
=\left(1-\theta\right)\left(p_t^*-p_t\right)+\left(1-\theta\right)\left(p_t-p_{t-1}\right) \\[-1pt]
&\hspace{4.1em}
=\left(1-\theta\right)\left(p_t^*-p_t\right)+\left(1-\theta\right)\pi_t \\[-1pt]
&\Rightarrow\quad
p_t^*-p_t=\frac{\theta}{1-\theta}\pi_t
\Rightarrow
S_t^*=\frac{\theta}{1-\theta}\pi_t
\qquad \purp{\text{price-setter dispersion.}}
\end{align*}
\[
\text{Non-price-resetters have}\quad
\left\{
\begin{aligned}
S_t\left(i\right)&=p_t\left(i\right)-p_t,\\[-1pt]
S_{t-1}\left(i\right)&=p_{t-1}\left(i\right)-p_{t-1},\\[-1pt]
p_t\left(i\right)&=p_{t-1}\left(i\right).
\end{aligned}\right.
\]
\[
\Rightarrow S_t\left(i\right)-S_{t-1}\left(i\right)
=
\left[p_t\left(i\right)-p_t\right]-\left[p_{t-1}\left(i\right)-p_{t-1}\right]
=
-p_t+p_{t-1}
=
-\pi_t.
\]
\[
\Rightarrow\quad
S_t\left(i\right)=S_{t-1}\left(i\right)-\pi_t.
\]
\begin{align*}
V_t&=\left(1-\theta\right)\left(S_t^*\right)^2+\theta\int_0^1\left(S_{t-1}\left(i\right)-\pi_t\right)^2di\\[-1pt]
&=\frac{\theta^2}{1-\theta}\pi_t^2+\theta V_{t-1}+\theta\pi_t^2
\qquad\purp{\text{We show it at the end}}\\[-1pt]
&=\theta V_{t-1}+\frac{\theta}{1-\theta}\pi_t^2.
\end{align*}
Assume $V_{-1}=0$:
\begin{align*}
\sum_{t=0}^{\infty}\beta^tV_t
&=\theta\sum_{t=0}^{\infty}\beta^tV_{t-1}
+\frac{\theta}{1-\theta}\sum_{t=0}^{\infty}\beta^t\pi_t^2,\\[-1pt]
\text{Since }V_{-1}=0,
\quad \sum_{t=0}^{\infty}\beta^tV_{t-1}
&=\beta\sum_{t=1}^{\infty}\beta^{t-1}V_{t-1}+V_{-1}
=\beta\sum_{t=0}^{\infty}\beta^tV_t,\\[-1pt]
\Rightarrow\quad
\sum_{t=0}^{\infty}\beta^tV_t
&=\beta\theta\sum_{t=0}^{\infty}\beta^tV_t
+\frac{\theta}{1-\theta}\sum_{t=0}^{\infty}\beta^t\pi_t^2,\\[-1pt]
\Rightarrow\quad
\left(1-\beta\theta\right)\sum_{t=0}^{\infty}\beta^tV_t
&=\frac{\theta}{1-\theta}\sum_{t=0}^{\infty}\beta^t\pi_t^2,\\[-1pt]
\Rightarrow\quad
\sum_{t=0}^{\infty}\beta^tV_t
&=\frac{\theta}{\left(1-\theta\right)\left(1-\beta\theta\right)}\sum_{t=0}^{\infty}\beta^t\pi_t^2.\tag{3}
\end{align*}

\par\vspace{0.6em}

\[
\frac{MPN_t\Delta N_t}{Y_t}=\frac12\frac{\varepsilon}{\Theta}V_t
\Rightarrow
\text{Total present value of welfare loss}
=\frac12\frac{\varepsilon}{\Theta}\sum_{t=0}^{\infty}\beta^tV_t.
\]
Plug in (3):
\[
L^\pi=\frac12\frac{\varepsilon}{\Theta}
\frac{\theta}{\left(1-\theta\right)\left(1-\beta\theta\right)}
\sum_{t=0}^{\infty}\beta^t\pi_t^2.
\]
Define
\[
\lambda=\frac{\left(1-\theta\right)\left(1-\beta\theta\right)}{\theta}\Theta,
\qquad \Theta\text{ same as lecture 8.}
\]
\[
L^\pi=\frac12\frac{\varepsilon}{\lambda}\sum_{t=0}^{\infty}\beta^t\pi_t^2
=\frac12\sum_{t=0}^{\infty}\beta^t\frac{\varepsilon}{\lambda}\hat\pi_t^2.
\]
\[
L^C+L^N
=\frac12\sum_{t=0}^{\infty}\beta^t
\left(\sigma+\frac{\alpha+\psi}{1-\alpha}\right)\tilde y_t^2.
\]
Thus
\[
L=L^C+L^N+L^\pi
=\frac12\sum_{t=0}^{\infty}\beta^t
\left[\left(\sigma+\frac{\alpha+\psi}{1-\alpha}\right)\tilde y_t^2
+\frac{\varepsilon}{\lambda}\hat\pi_t^2\right].
\]
Why
\[
\theta\int_0^1\left(S_{t-1}\left(i\right)-\pi_t\right)^2\,di
=
\theta V_{t-1}+\theta\pi_t^2 .
\]

\begin{align*}
\int_0^1\left(S_{t-1}\left(i\right)-\pi_t\right)^2\,di
&=
\int_0^1 S_{t-1}\left(i\right)^2\,di
+\int_0^1 \pi_t^2\,di
-2\pi_t\int_0^1 S_{t-1}\left(i\right)\,di \\[2pt]
&=
V_{t-1}
+\pi_t^2
-2\pi_t\int_0^1 S_{t-1}\left(i\right)\,di .
\end{align*}

Since we have
\[
\int_0^1 S_{t-1}\left(i\right)\,di
=
\frac{\varepsilon-1}{2}V_{t-1},
\]

\begin{align*}
\int_0^1\left(S_{t-1}\left(i\right)-\pi_t\right)^2\,di
&=
V_{t-1}
+\pi_t^2
-2\pi_t\cdot
\frac{\varepsilon-1}{2}V_{t-1} \\[2pt]
&=
V_{t-1}
+\pi_t^2
-\left(\varepsilon-1\right)V_{t-1}\pi_t .
\end{align*}
\green{$\pi_t$ is first-order, $V_{t-1}$ is second-order $\Rightarrow \pi_tV_{t-1}$ is third-order.}

\green{In a second-order welfare approximation, we drop third-order terms.}
\[
\Rightarrow\quad
\theta\int_0^1\left(S_{t-1}\left(i\right)-\pi_t\right)^2di=\theta V_{t-1}+\theta\pi_t^2.
\]

From
\[
L=\frac12\sum_{t=0}^{\infty}\beta^t
\left[\left(\sigma+\frac{\alpha+\psi}{1-\alpha}\right)\tilde y_t^2
+\frac{\varepsilon}{\lambda}\hat\pi_t^2\right],
\]
where
\[
\lambda=\frac{\left(1-\theta\right)\left(1-\beta\theta\right)}{\theta}\Theta ,
\qquad
\Theta=\frac{1-\alpha}{1-\alpha+\alpha\varepsilon}.
\]
\[
\lambda\text{ is the slope of NKPC with respect to real marginal cost.}
\]
Let
\[
\kappa=\lambda\left(\sigma+\frac{\psi+\alpha}{1-\alpha}\right)
\quad\cdots\quad
\text{slope of NKPC with respect to }\tilde y_t.
\]
\[
L
=
\frac12\sum_{t=0}^{\infty}\beta^t
\left[
\frac{\kappa}{\lambda}\tilde y_t^2
+
\frac{\varepsilon}{\lambda}\hat\pi_t^2
\right]
=
\frac12\frac{\varepsilon}{\lambda}
\sum_{t=0}^{\infty}\beta^t
\left[
\frac{\kappa}{\varepsilon}\tilde y_t^2
+
\hat\pi_t^2
\right].
\]

\par\vspace{0.6em}
We let
\[
\alpha_x=\frac{\kappa}{\varepsilon}\qquad
L=\frac12\frac{\varepsilon}{\lambda}\sum_{t=0}^{\infty}\beta^t
\left[\hat\pi_t^2+\alpha_x\tilde y_t^2\right].
\]
When we have inefficiency at supply side:
\[
\tilde{y}_t
=
\hat{y}_t-\hat{y}_t^n
\neq
\hat{y}_t-\hat{y}_t^e
=
\hat{x}_t .
\]
The welfare loss is coming from the gap between realized current output $\hat y_t$ \purp{(deviation)} and the efficient output $\hat y_t^e$ \purp{(deviation)}, thus
\[
L=\frac12\frac{\varepsilon}{\lambda}\sum_{t=0}^{\infty}\beta^t
\left[\hat\pi_t^2+\alpha_x\hat x_t^2\right].
\]
Now scale the welfare loss by $2\frac{\lambda}{\varepsilon}$:
\[
L'\equiv2\frac{\lambda}{\varepsilon}L
=\sum_{t=0}^{\infty}\beta^t
\left[\hat\pi_t^2+\alpha_x\hat x_t^2\right].
\]
since $2\cdot\dfrac{\lambda}{\varepsilon}$ is constant, the scaling does not affect the optimal decision $\left(\hat\pi_t^*,\hat x_t^*\right)$ that minimizes the loss $L'$.
