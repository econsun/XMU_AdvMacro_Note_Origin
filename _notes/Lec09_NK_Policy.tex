\section{Efficient allocation and distortions}

\[
\text{Monopolistic competition + flexible prices}
\Rightarrow
\begin{aligned}[t]
&\text{efficient allocation, with a subsidy to correct distortions}\\
&\text{of monopolistic competition.}
\end{aligned}
\]
\[
\text{Sticky prices + policy that fully stabilizes the price level}
\Rightarrow
\text{efficient allocation.}
\]

\section{Efficient allocation under social planner's problem}

\[
\max_{\left\{C_t,N_t\right\}} U\left(C_t,N_t\right)
\]
\[
\text{s.t.}\qquad C_t\left(i\right)=A_tF\left(N_t\left(i\right)\right),
\quad \text{for all }i\in\left[0,1\right],
\]
\[
C_t=
\left(\int_0^1 C_t\left(i\right)^{\frac{\varepsilon-1}{\varepsilon}}\dd i\right)^{\frac{\varepsilon}{\varepsilon-1}},
\qquad
N_t=\int_0^1N_t\left(i\right)\dd i.
\]
\[
\begin{alignedat}{3}
C_t\left(i\right) &= C_t,
&\qquad &\forall i\in\left[0,1\right],
&\qquad &\left(1\right)
\\
N_t\left(i\right) &= N_t,
&\qquad &\forall i\in\left[0,1\right],
&\qquad &\left(2\right)
\\
-\frac{U_{N,t}}{U_{C,t}}
&=MPN_t
=\left(1-\alpha\right)A_tN_t^{-\alpha},
&&
&\qquad &\left(3\right)
\end{alignedat}
\]
\[
\begin{array}{@{}l@{\quad}c@{\quad}l@{}}
\text{nominal rigidity}
&\Rightarrow&
\left\{
\begin{aligned}
C_t\left(i\right)&\neq C_t,\\
N_t\left(i\right)&\neq N_t,
\end{aligned}
\right.
\\[1.0em]
\text{nominal rigidity or monopolistic competition}
&\Rightarrow&
\text{(3) does not hold.}
\\[1.0em]
\textcolor{red!70!black}{\text{(1)\&(2) symmetric allocation}}
&\Rightarrow&
MPN_t=MPN_t\left(i\right).
\end{array}
\]
Optimality conditions can be met due to:
\begin{enumerate}[label=\roman*.]
  \item All goods enter utility symmetrically.\par
  \red{For planner, weight are the same even distribution }$C_t\left(i\right)$.
  \item Identical technology of production.\par
  \red{For planner, technology for $i$ is the same, even distribution }$N_t\left(i\right)$.
  \item Concavity of utility.\par
  \red{For planner, there is an optimal allocation of $C_t\left(i\right)$ and $N_t\left(i\right)$.}
\end{enumerate}

\[
\MRS_{C,N}
=
-\frac{U_{N,t}}{U_{C,t}},
\qquad
\MRT_{C,N}
=
\frac{\dd C_t}{\dd N_t}
=
F_N\left(N_t\right)A_t
\quad
\red{\text{output transfered one-for-one to consumption.}}
\]
\[
\red{\Rightarrow\MRS_{C,N}=\MRT_{C,N}=\MPN.}
\]

\section{Suboptimality in NK}

Two sources of distortion:
\begin{enumerate}[label=\arabic*)]
  \item Monopolistic competitive firms $\Rightarrow$ market power.
  \item Sticky price.
\end{enumerate}

Let's study (1) separately. Assume flexible price. With isoelastic demand function, optimal pricing:
\[
P_t
=
\mu\frac{W_t}{\MPN_t},
\qquad
\mu=\frac{\varepsilon}{\varepsilon-1}
\quad\text{markup}.
\]
\[
\Rightarrow
\MRS_{C,N}
=
-\frac{U_{N,t}}{U_{C,t}}
=
\frac{W_t}{P_t}
=
\frac{\MPN_t}{\mu}.
\]
\[
\red{\mu>1\Rightarrow \text{low employment and output, and optimal condition }\left(3\right)\text{ is violated.}}
\]

Let $\tau$ denotes the employment subsidy; subsidy financed by lump-sum tax, \red{no distortions}.
\[
P_t
=
\mu\frac{\left(1-\tau\right)W_t}{\MPN_t}
\Rightarrow
-\frac{U_{N,t}}{U_{C,t}}
=
\frac{W_t}{P_t}
=
\frac{\MPN_t}{\mu\left(1-\tau\right)}.
\]
\[
\MPN_t
=
\frac{\MPN_t}{\mu\left(1-\tau\right)}
\Rightarrow
\mu\left(1-\tau\right)=1
\Rightarrow
\tau=\frac{1}{\varepsilon}.
\]
\[
\text{The optimal allocation can be achieved by giving employment subsidy of }\tau=\frac{1}{\varepsilon}.
\]

\par\smallskip
Now study (2) price stickiness.
\subsection{Average markup varies over time}

\[
\mu_t
=
\frac{P_t}{\left(1-\tau\right)W_t/\MPN_t}
=
\frac{\mu P_t}{W_t/\MPN_t}
\Rightarrow
\frac{W_t}{P_t}=\frac{\mu}{\mu_t}\MPN_t.
\]
\[
-\frac{U_{N,t}}{U_{C,t}}
=
\frac{W_t}{P_t}
=
\frac{\mu}{\mu_t}\MPN_t.
\]
\[
\text{If }\mu_t\ne\mu,
\quad\text{optimal allocation condition }\left(3\right)\text{ is violated.}
\]

\subsection{Staggered price setting: sticky price}

\[
P_t\left(i\right)\ne P_t\left(j\right),
\qquad
C_t\left(i\right)\ne C_t\left(j\right),
\qquad
N_t\left(i\right)\ne N_t\left(j\right).
\]
\[
\text{Optimal conditions }\left(1\right)\text{ and }\left(2\right)\text{ are violated.}
\]

\section{Optimal monetary policy in the basic NK}

Assumptions:
\begin{enumerate}[label=\arabic*.]
  \item An optimal subsidy to offset distortion of firms' market power.
  \item Economy starts without relative price distortion:
  \[
  P_{-1}\left(i\right)=P_{-1},
  \qquad \text{for all }i\in\left[0,1\right].
  \]
\end{enumerate}

Optimal policy should achieve:
\begin{enumerate}[label=\arabic*.]
  \item $P_t=P_{t-1}$, $\forall t=0,1,2,\ldots$, so that conditions (1) and (2) hold.
  \item $\mu_t=\mu$, $\forall t$, so that condition (3) holds.
\end{enumerate}

\[
\Leftrightarrow
\tilde y_t=\hat y_t-\hat y_t^n=0,
\quad \forall t;
\qquad
\hat\pi_t=0,
\quad \forall t.
\]

DIS:
\[
\tilde y_t
=
-\frac{1}{\sigma}\left(\hat i_t-\E_t\left\{\hat\pi_{t+1}\right\}-\hat r_t^n\right)
+
\E_t\left\{\tilde y_{t+1}\right\}.
\]
\[
\Rightarrow
\hat i_t=\hat r_t^n,
\qquad \forall t.
\]

\textbf{1. Stabilizing output:} \red{Not freezing output at constant.}
\[
\hat y_t=\hat y_t^n,
\qquad \forall t.
\]
\[
\purple{\text{Eliminating gap, not eliminating fluctuations.}}
\]
\[
\blue{\hat y_t^n\text{ is subject to technology shock.}}
\]

\textbf{2. Stabilizing price:} divine coincidence.
\[
\text{If central bank achieves price stability, then it automatically gets }\tilde y_t=0.
\]
\[
\purple{\text{Stabilizing inflation }\Rightarrow\text{ stabilizing output gap.}}
\]

Non-policy block:

Policy block:
\par\smallskip
\noindent
\textbf{Case a: An exogenous interest rate rule
}
\[
\hat i_t=\hat r_t^n,
\qquad \forall t.
\]
\[
\Rightarrow
\left\{
\begin{aligned}
\boxed{\tilde y_t}
&=
\frac{1}{\sigma}E_t\left\{\hat\pi_{t+1}\right\}
+
E_t\left\{\tilde y_{t+1}\right\},
\\
\hat\pi_t
&=
\beta E_t\left\{\hat\pi_{t+1}\right\}
+
\kappa \tilde y_t
\end{aligned}
\right.
\]

Substitute the first equation into the second:
\begin{align*}
\hat\pi_t
&=\beta E_t\left\{\hat\pi_{t+1}\right\}
+\kappa\left(\dfrac{1}{\sigma}E_t\left\{\hat\pi_{t+1}\right\}+E_t\left\{\tilde y_{t+1}\right\}\right)\\
&=\kappa E_t\left\{\tilde y_{t+1}\right\}
+\left(\beta+\dfrac{\kappa}{\sigma}\right)E_t\left\{\hat\pi_{t+1}\right\}.
\end{align*}
\[
\Rightarrow
\begin{bmatrix}
\tilde y_t\\
\hat\pi_t
\end{bmatrix}
=
A_0
\begin{bmatrix}
E_t\left\{\tilde y_{t+1}\right\}\\
E_t\left\{\hat\pi_{t+1}\right\}
\end{bmatrix},
\qquad
\text{where }
A_0=
\begin{bmatrix}
1 & \frac{1}{\sigma}\\
\kappa & \beta+\frac{\kappa}{\sigma}
\end{bmatrix}.
\]

\[
\purple{\tilde y_t=\hat\pi_t=0\text{ is one solution associated with optimal policy, but not unique.}}
\]
\[
\purple{\tilde y_t\text{ and }\hat\pi_t\text{ are neither predetermined.}}
\]

\[
\det\left(A_0\right)=\beta,
\qquad
\operatorname{tr}\left(A_0\right)=1+\beta+\frac{\kappa}{\sigma}.
\]
\[
P\left(1\right)=1-T+D=1-1-\beta-\frac{\kappa}{\sigma}+\beta=-\frac{\kappa}{\sigma}<0.
\]
\[
P\left(-1\right)=1+T+D=1+1+\beta+\frac{\kappa}{\sigma}+\beta
=2+2\beta+\frac{\kappa}{\sigma}>0.
\]
\[
\text{There is one root between }\left(-1,1\right)\text{ and another one outside the unit circle.}
\]
\[
\purple{\#\text{ jump variables } > \#\text{ unstable roots }\Rightarrow\text{ system is indeterminate.}}
\]
\[
\purple{\text{No guarantee on the realization of }\tilde y_t=\hat\pi_t=0,\quad \forall t.}
\]
\par\smallskip
\noindent
\textbf{Case b: A Taylor-type interest rate rule}


\[
\hat i_t
=
\hat r_t^n+\phi_\pi\hat\pi_t+\phi_y\tilde y_t,
\qquad
\phi_\pi\ge0,
\quad
\phi_y\ge0,
\quad \forall t.
\]

\[
\left\{
\begin{aligned}
\tilde y_t
&=
-\frac{\phi_\pi}{\sigma}\hat\pi_t
-
\frac{\phi_y}{\sigma}\tilde y_t
+
\frac{1}{\sigma}\E_t\left\{\hat\pi_{t+1}\right\}
+
\E_t\left\{\tilde y_{t+1}\right\},\\
\hat\pi_t
&=
\beta\E_t\left\{\hat\pi_{t+1}\right\}
+
\kappa\tilde y_t.
\end{aligned}
\right.
\]

\[
\left\{
\begin{aligned}
\tilde y_t
&=
\frac{1-\beta\phi_\pi}{\sigma+\phi_y+\kappa\phi_\pi}
\E_t\left\{\hat\pi_{t+1}\right\}
+
\frac{\sigma}{\sigma+\phi_y+\kappa\phi_\pi}
\E_t\left\{\tilde y_{t+1}\right\},\\
\hat\pi_t
&=
\frac{\kappa+\beta\sigma+\beta\phi_y}{\sigma+\phi_y+\kappa\phi_\pi}
\E_t\left\{\hat\pi_{t+1}\right\}
+
\frac{\sigma\kappa}{\sigma+\phi_y+\kappa\phi_\pi}
\E_t\left\{\tilde y_{t+1}\right\}.
\end{aligned}
\right.
\]

\[
\begin{bmatrix}
\tilde y_t\\
\hat\pi_t
\end{bmatrix}
=
A_T
\begin{bmatrix}
\E_t\left\{\tilde y_{t+1}\right\}\\
\E_t\left\{\hat\pi_{t+1}\right\}
\end{bmatrix},
\]
\[
A_T
=
\Omega
\begin{bmatrix}
\sigma & 1-\beta\phi_\pi\\
\kappa\sigma & \kappa+\beta\left(\sigma+\phi_y\right)
\end{bmatrix},
\qquad
\Omega=\frac{1}{\sigma+\phi_y+\kappa\phi_\pi}.
\]

\[
\tilde y_t=\hat\pi_t=0,\quad \forall t
\quad
\text{is always a solution.}
\]

\purple{
To guarantee uniqueness, we need two stable roots
\[
\text{notice that } x_t=A_T E_tx_{t+1},
\quad
\text{but BK is defined on }
E_tx_{t+1}=A_T^{-1}x_t,
\]
\[
\text{2 stable roots in } A_T
\Longleftrightarrow
\text{2 unstable roots in } A_T^{-1}.
\]
}

\[
\operatorname{tr}\left(A_T\right)=\Omega\left[\sigma+\kappa+\beta\left(\sigma+\phi_y\right)\right],
\qquad
\det\left(A_T\right)=\Omega\beta\sigma.
\]
\[
\text{To have stable result, need }D<1,
\quad P\left(1\right)>0,
\quad P\left(-1\right)>0.
\]
\[
\begin{aligned}
P\left(1\right)
&=
1-\operatorname{tr}\left(A_T\right)+\det\left(A_T\right)
=
1-\Omega\left[\sigma+\kappa+\beta\phi_y\right]
>0\\
&\Rightarrow
\sigma+\kappa+\beta\phi_y
<
\sigma+\phi_y+\kappa\phi_\pi\\
&\Rightarrow
\kappa\left(\phi_\pi-1\right)+\left(1-\beta\right)\phi_y>0
\qquad \left(*\right)
\end{aligned}
\]

\[
P\left(-1\right)
=
1+\operatorname{tr}\left(A_T\right)+\det\left(A_T\right)
=
1+\Omega\left[\sigma+\kappa+\beta\phi_y+\beta\sigma\right]
>0
\quad
\text{for sure.}
\]

\[
\Delta
=
\operatorname{Tr}\left(A_T\right)^2
-
4\det\left(A_T\right)
=
\Omega^2\left[\sigma+\kappa+\beta\left(\sigma+\phi_y\right)\right]^2
-
4\Omega\beta\sigma,
\qquad
\text{sign does not matter.}
\]
\[
\begin{aligned}
D
&=
\det\left(A_T\right)
=
\Omega\beta\sigma
=
\frac{\beta\sigma}{\sigma+\phi_y+\kappa\phi_\pi}
=
\frac{1}{
\frac{1}{\beta}
+
\frac{\phi_y+\kappa\phi_\pi}{\beta\sigma}
}
\\
&<
\frac{1}{
1+
\frac{\kappa-\kappa\phi_\pi+\kappa\phi_\pi}{\beta\sigma}
}
=
\frac{1}{
1+\frac{\kappa}{\beta\sigma}
}
<1.
\end{aligned}
\]

\[
\text{If }\kappa\left(\phi_\pi-1\right)+\left(1-\beta\right)\phi_y>0\text{ is satisfied,}
\quad
\tilde y_t=\hat\pi_t=0
\quad\text{and}\quad
\hat i_t=\hat r_t^n
\text{ hold}

\]
\[
\text{If }\phi_y=0
\Rightarrow
\kappa\left(\phi_\pi-1\right)>0

\phi_\pi>1
\quad
\text{Taylor principle.}
\]
\[
\text{If }\phi_y>0,
\quad
\text{output gap helps, lower bound on }\phi_\pi\text{ can be relaxed a little.}
\]
\[
\purple{\text{But }1-\beta\text{ is usually small, inflation is the main stabilizing force.}}
\]\par\smallskip

\section{Simple monetary policy rule}
\subsection{Rotemberg and Woodford (1999)}

Second-order approximation to the utility losses due to deviations from efficient allocation:
\[
W
=
\frac{1}{2}\E_0\sum_{t=0}^{\infty}\beta^t
\left[
\left(\sigma+\frac{\varphi+\alpha}{1-\alpha}\right)\tilde y_t^2
+
\frac{\varepsilon}{\kappa}\hat\pi_t^2
\right].
\]

Per-period loss:
\[
L
=
\frac{1}{2}
\left[
\left(\sigma+\frac{\varphi+\alpha}{1-\alpha}\right)\Var\left(\tilde y_t\right)
+
\frac{\varepsilon}{\kappa}\Var\left(\hat\pi_t\right)
\right].
\]
\[
\sigma\uparrow,\ \varphi\uparrow,\ \alpha\uparrow
\Rightarrow L\uparrow.
\quad
\text{Curvature amplifies the effect (consumption, labor, technology).}
\]
\[
\varepsilon\uparrow,\ \frac{1}{\kappa}\uparrow
\Rightarrow L\uparrow.
\quad
\text{Amplify losses from price dispersion; }\frac{1}{\lambda}\text{ amplifies the degree of price dispersion.}
\]

\begin{bmatrix}

A_T
\begin{bmatrix}
\E_t\left\{\tilde y_{t+1}\right\}\\
