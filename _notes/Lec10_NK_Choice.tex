\textbf{Myopic optimiser} & \textbf{Dynamic optimiser} \\
Optimize period by period. & Optimize the sum of present and future utilities. \\
\purple{discretion: period-by-period optimisation without committing to future actions.} &
\purple{commitment: central bank can credibly announce and follow a future path. Backward induction.}

\par\smallskip
Recap: Optimal Monetary Policy is strict inflation targeting.
\[
\hat\pi_t=0
\Rightarrow
\hat y_t=\hat y_t^{\,n}=\hat y_t^{\,e}.
\]
However, possible and optimal but not feasible
\red{since central bank can not fully stabilize inflation.}

\red{Facing short-run tradeoff due to supply shock.}

\section{NK with imperfection at the supply side}

Under flexible price:
\[
y_t^{\,n}\neq y_t^{\,e}
\qquad
(\text{natural output} \neq \text{efficient output})
\Rightarrow
\text{short-run tradeoff between quantity and prices.}
\]
\purple{Move oppositely facing supply shock.}

Assume at steady state, the inefficiency associated with the flexible price is not there:
\[
\bar y=\bar y^{\,n}=\bar y^{\,e},
\]
while exists short-run deviations:
\[
\hat y_t\ne \hat y_t^{\,n}\ne \hat y_t^{\,e}.
\]

Welfare losses are:
\[
E_0\sum_{t=0}^{\infty}\beta^t
\left(\hat\pi_t^2+\alpha_x\hat x_t^2\right).
\]

Define welfare relevant output gap as
\begin{align*}
\hat x_t
&=\hat y_t-\hat y_t^{\,e}
=\log y_t-\log\bar y-\bigl(\log y_t^{\,e}-\log\bar y^{\,e}\bigr)
=\log y_t-\log y_t^{\,e},\\
\hat\pi_t&=\log P_t-\log P_{t-1},
\qquad
\alpha_x=\frac{\kappa}{\varepsilon}.
\end{align*}

\[
\left.
\begin{aligned}
\tilde y_t
&=\hat y_t-\hat y_t^{\,n}
=\hat x_t+(\hat y_t^{\,e}-\hat y_t^{\,n}),\\
\hat\pi_t
&=\beta E_t\{\hat\pi_{t+1}\}+\kappa\tilde y_t
\end{aligned}
\right\}
\Rightarrow
\hat\pi_t
=\beta E_t\{\hat\pi_{t+1}\}+\kappa\hat x_t+u_t,
\]
where
\[
v_t=\beta E_t\{\hat\pi_{t+1}\}+u_t.
\]
\par\vspace{0.3em}
\noindent
\begin{itemize}
    \item $E_t\{\hat\pi_{t+1}\}$: taken as given when choosing current-period policy.
    \item $u_t$: exogenous cost-push shock.
\end{itemize}

\noindent
A nonzero $u_t$ makes it impossible to attain simultaneously zero inflation and an efficient level of output. 
For example, $u_t$ may come from exogenous changes in desired wage markups, price markups, labor income taxes, minimum wages, etc.

Assume $u_t$ follows:
\[
u_t=\rho_u u_{t-1}+\varepsilon_t^{u},
\qquad \rho_u\in[0,1).
\]
\[
\begin{array}{ll}
\text{NKPC:}
&
\hat\pi_t
=
\beta E_t\{\hat\pi_{t+1}\}
+\kappa\hat x_t
+u_t,
\\[0.7em]
\text{DIS:}
&
\hat x_t
=
-\dfrac{1}{\sigma}
\left(
\hat i_t
-
E_t\{\hat\pi_{t+1}\}
-
\hat r_t^{\,e}
\right)
+
E_t\{\hat x_{t+1}\}.
\end{array}
\]

\[
\hat r_t^{\,e}
\text{ is the efficient real interest rate that supports the efficient allocation.}
\]

\section{Optimal policy under discretion (period-by-period optimization)}

\[
\min_{\hat x_t,\hat\pi_t}\quad \hat\pi_t^2+\alpha_x\hat x_t^2
\qquad
\text{s.t.}\quad
\hat\pi_t=\kappa\hat x_t+v_t,
\]
where
\[
v_t=\beta E_t\{\hat\pi_{t+1}\}+u_t.
\]

\noindent
$E_t\{\hat\pi_{t+1}\}$: independent of current policy; \quad
$u_t$: exogenous.

\[
\mathcal L
=\hat\pi_t^2+\alpha_x\hat x_t^2
-\lambda(\hat\pi_t-\kappa\hat x_t-v_t).
\]
\begin{align*}
\frac{\partial\mathcal L}{\partial \hat\pi_t}=0
&\Rightarrow 2\hat\pi_t=\lambda,\\
\frac{\partial\mathcal L}{\partial \hat x_t}=0
&\Rightarrow 2\alpha_x\hat x_t=-\lambda\kappa
\Rightarrow
\hat x_t=-\frac{\kappa}{\alpha_x}\hat\pi_t,
\qquad \forall t
\quad \red{\text{targeting rule}.}
\end{align*}

To dampen the rise in inflation due to cost-push shock,
\[
\hat x_t<0
\Rightarrow
\hat y_t<\hat y_t^{\,e}.
\]
\[
\Rightarrow
\text{drive output below efficiency level; negative output gap }\Rightarrow\text{ policy trade-off.}
\]

Plug into NKPC:
\begin{align*}
\hat\pi_t
&=\beta E_t\{\hat\pi_{t+1}\}+\kappa\hat x_t+u_t
=\beta E_t\{\hat\pi_{t+1}\}-\frac{\kappa^2}{\alpha_x}\hat\pi_t+u_t,\\
\Rightarrow\quad
\hat\pi_t
&=\frac{\alpha_x\beta}{\alpha_x+\kappa^2}E_t\{\hat\pi_{t+1}\}
+\frac{\alpha_x}{\alpha_x+\kappa^2}u_t.
\end{align*}

Iterate forward:
\[
\left\{
\begin{aligned}
\hat\pi_t&=\alpha_x\psi u_t,\\
\hat x_t&=-\kappa\psi u_t,
\end{aligned}
\right.
\qquad
\psi=\frac{1}{\kappa^2+\alpha_x(1-\beta\rho_u)}.
\]

Central bank lets output gap and inflation deviate from targets in proportion to the current value of the cost-push shock:
\[
\hat\pi_t=\alpha_x\psi u_t,
\qquad
\hat x_t=-\kappa\psi u_t.
\]

\[
{\color{noteRed}
\psi_{0.5}
=
\frac{1}{\kappa^2+\alpha_x\left(1-\frac{\beta}{2}\right)}
>
\psi_0
=
\frac{1}{\kappa^2+\alpha_x}.
}
\]

\purple{Optimal to partly accommodate inflation pressure from the cost-push shock by letting inflation rise.}

Without any policy response, $\kappa=0$:
\[
\hat\pi_t=\beta E_t\{\hat\pi_{t+1}\}+u_t.
\]
Guess $\hat\pi_t=Au_t$ and $E_tu_{t+1}=\rho_u u_t$:
\[
Au_t=\beta A\rho_u u_t+u_t
\Rightarrow
A=\frac{1}{1-\beta\rho_u}.
\]
Without output gap response:
\[
\hat\pi_t=\frac{1}{1-\beta\rho_u}u_t
>
\frac{\alpha_x}{\kappa^2+\alpha_x({1-\beta\rho_u})}u_t.
\]
\[
\begin{aligned}
\hat x_t
&=
-\kappa\psi u_t
\Rightarrow
\kappa\hat x_t
=
-\kappa^2\psi u_t<0
\quad \text{when }u_t>0,\\
\hat\pi_t
&=
\beta E_t\hat\pi_{t+1}
\boxed{-\kappa^2\psi u_t}
+
u_t.
\end{aligned}
\]

\par\smallskip
\noindent
\purple{$u_t$ pushes inflation up --- direct effect.}

\noindent
\purple{Induced recessionary output gap pushes inflation down.}

\noindent
Using
\[
E_t\{\hat\pi_{t+1}\}=\alpha_x\psi\rho_u u_t,
\]
\[
\hat\pi_t
=\left[\beta\alpha_x\psi\rho_u-\kappa^2\psi+1\right]u_t
=\alpha_x\psi u_t.
\]
\[
\hat\pi_t=\hat p_t-\hat p_{t-1}
\Rightarrow
\hat p_t=\hat p_{t-1}+\sum_{j=0}^{t}\hat\pi_j.
\]

\[
\text{Since } \hat\pi_j=\alpha_x\psi u_j,
\quad
\text{and } u_j\to0 \text{ as } j\to\infty,
\Rightarrow
\hat\pi_j\to0 \text{ as } j\to\infty.
\]

\[
\Rightarrow
\lim_{t\to\infty}\hat p_t
=
\hat p_{-1}
+
\sum_{j=0}^{\infty}\hat\pi_j
>
\hat p_{-1},
\qquad
\purple{\underline{\text{level price stay permanently higher.}}}
\]

\section{Optimal discretionary policy implementation}

\[
\hat\pi_t=\alpha_x\psi u_t,
\qquad
\hat x_t=-\kappa\psi u_t.
\tag{Target rule}
\]
\[
\hat x_t=-\frac{1}{\sigma}\left(\hat i_t-E_t\{\hat\pi_{t+1}\}-\hat r_t^{\,e}\right)+E_t\{\hat x_{t+1}\}.
\]
Plug in:
\begin{align*}
-\kappa\psi u_t
&=
-\frac{1}{\sigma}
\left(\hat i_t-\alpha_x\psi\rho_u u_t-\hat r_t^{\,e}\right)
-\kappa\psi\rho_u u_t,\\
\Rightarrow\quad
\hat i_t
&=
\underbrace{\hat r_t^{\,e}}_{\text{efficiency rate}}
+
\underbrace{\psi_i u_t}_{\text{response from cost-push shock}},
\qquad
\psi_i=\psi\left[\kappa\sigma(1-\rho_u)+\alpha_x\rho_u\right].
\end{align*}
\purple{$\kappa\sigma(1-\rho_u)$: need to affect current output gap through IS curve.}\
\purple{$\alpha_x\rho_u$: forward-looking part.}

\red{$u_t\uparrow$ cost-push shock $\Rightarrow$ inflation $\hat\pi_t\uparrow$ $\Rightarrow$ cool demand through $\hat i_t\uparrow>\hat r_t^{\,e}$ $\Rightarrow$ $\hat x_t\downarrow$ $\Rightarrow$ $\hat\pi_t\downarrow$.}\qquad
But output losses are costly $\Rightarrow$ only fights $\hat\pi_t\uparrow$ partially.

Consider an inflation targeting rule:
\[
\hat i_t=\hat r_t^{\,e}+\phi_\pi\hat\pi_t,
\qquad
\phi_\pi=(1-\rho_u)\frac{\kappa\sigma}{\alpha_x}+\rho_u.
\]
\[
\begin{array}{@{}l@{\qquad}l@{}}
\hat i_t=\hat r_t^{\,e}+\psi_i u_t,
&
\hat\pi_t=\alpha_x\psi u_t
\Rightarrow
u_t=\dfrac{\hat\pi_t}{\alpha_x\psi}
\\[0.25em]
&
\Rightarrow
\hat i_t
=
\hat r_t^{\,e}
+
\dfrac{\psi_i}{\alpha_x\psi}\hat\pi_t.
\end{array}
\]
\[
\frac{\psi_i}{\alpha_x\psi}
=\frac{\kappa\sigma(1-\rho_u)+\alpha_x\rho_u}{\alpha_x}
=(1-\rho_u)\frac{\kappa\sigma}{\alpha_x}+\rho_u
=\phi_\pi.
\]

Since a determinate equilibrium exists iff $\phi_\pi>1$:
\[
\phi_\pi>1
\Rightarrow
\frac{\kappa\sigma}{\alpha_x}>1
\quad \text{may not be satisfied.}
\]

A rule that decentralizes the desired discretionary equilibrium:
\begin{align*}
\hat i_t
&=\hat r_t^{\,e}+\psi_i u_t+\phi_\pi(\hat\pi_t-\alpha_x\psi u_t)\\
\tag{Instrument rule}
&=\hat r_t^{\,e}+\Theta_i u_t+\phi_\pi\hat\pi_t,
\end{align*}
where
\[
\Theta_i=\psi\left[\kappa\sigma(1-\rho_u)-\alpha_x(\phi_\pi-\rho_u)\right].
\]
Now we have a unique equilibrium, and it is the desired discretionary equilibrium.

\begin{enumerate}
  \item Under desired equilibrium:
  \[
  \hat\pi_t=\alpha_x\psi u_t,
  \qquad
  \hat i_t=\hat r_t^{\,e}+\psi_i u_t.
  \]
  \blue{We do not change the intended discretionary allocation.}
  \item Central bank reacts more aggressively:
  \[
  \hat i_t\uparrow\quad \text{when}\quad \hat\pi_t>\alpha_x\psi u_t.
  \]
  If $\phi_\pi>1$, nominal rate rises more than one-for-one with inflation deviations; real rate rises, demand falls, inflation falls, eliminating self-fulfilling inflation paths.
\end{enumerate}

\[
\begin{aligned}
\purple{\text{Target rule:}}\quad
&\purple{\text{tell central bank the relationship between target variables to achieve.}}\\
\purple{\text{E.g.}}\quad
&\hat x_t=-\frac{\kappa}{\alpha_x}\hat\pi_t,
\qquad
\purple{\text{This tells tradeoff, but still infeasible, since } y_t^e \text{ is unobservable.}}
\end{aligned}
\]

\[
\begin{aligned}
\purple{\text{Instrument rule:}}\quad
&\purple{\text{tell central bank how to set its instrument.}}\\
\purple{\text{E.g.}}\quad
&\hat i_t=\hat r_t^{\,e}+\Theta_i u_t+\phi_\pi\hat\pi_t,
\qquad
\purple{\hat r_t^{\,e} \text{ and } u_t \text{ are hard to observe and measure.}}
\end{aligned}
\]
\section{Optimal policy under commitment}

Central bank able to commit with full credibility to a policy plan.
\[
\min_{\{\hat x_t,\hat\pi_t\}_{t=0}^{\infty}}
\frac12 E_0\sum_{t=0}^{\infty}\beta^t
\left(\hat\pi_t^2+\alpha_x\hat x_t^2\right)
\]
\[
\text{s.t.}\quad
\hat\pi_t=\beta E_t\{\hat\pi_{t+1}\}+\kappa\hat x_t+u_t,
\qquad
u_t=\rho_u u_{t-1}+\varepsilon_t^u.
\]

\[
\mathcal L
=E_0\sum_{t=0}^{\infty}\beta^t
\left[\frac12(\hat\pi_t^2+\alpha_x\hat x_t^2)
+\gamma_t(\hat\pi_t-\kappa\hat x_t-\beta E_t\{\hat\pi_{t+1}\}-u_t)
\right].
\]

F.O.C.:
\begin{align*}
\frac{\partial\mathcal L}{\partial \hat\pi_t}=0
&\Rightarrow
\hat\pi_t+\gamma_t-\gamma_{t-1}=0,
\qquad \gamma_{-1}=0,\\
\frac{\partial\mathcal L}{\partial \hat x_t}=0
&\Rightarrow
\alpha_x\hat x_t-\kappa\gamma_t=0
\Rightarrow
\gamma_t=\frac{\alpha_x}{\kappa}\hat x_t,
\qquad \forall t.
\end{align*}
\purple{Constraint in period $-1$ not effective for optimal plan in period $0$.}
\[
\Rightarrow
\left\{
\begin{aligned}
\hat x_0
&=
-\frac{\kappa}{\alpha_x}\hat\pi_0,\\
\hat x_t
&=
\hat x_{t-1}
-
\frac{\kappa}{\alpha_x}\hat\pi_t,
\qquad t\ge 1.
\end{aligned}
\right.
\qquad
\purple{(*)}
\Rightarrow
\hat x_t
=
-\frac{\kappa}{\alpha_x}\check p_t.
\]

\[
\begin{aligned}
\hat x_1
&=
\hat x_0-\frac{\kappa}{\alpha_x}\hat\pi_1
=
-\frac{\kappa}{\alpha_x}
\left(
\hat\pi_0+\hat\pi_1
\right),
\\
\hat x_2
&=
\hat x_1-\frac{\kappa}{\alpha_x}\hat\pi_2
=
-\frac{\kappa}{\alpha_x}
\left(
\hat\pi_0+\hat\pi_1+\hat\pi_2
\right).
\end{aligned}
\]

\[
\Rightarrow
\hat x_t
=
-\frac{\kappa}{\alpha_x}
\sum_{s=0}^{t}\hat\pi_s.
\]

\[
\because
\quad
\hat\pi_s=\log P_s-\log P_{s-1},
\qquad

\sum_{s=0}^{t}
\left(
\log P_s-\log P_{s-1}
\right)

\]


\[
\quad\Rightarrow\quad
\hat x_t
=
-\frac{\kappa}{\alpha_x}\check p_t.
\qquad
\check p_t=\log P_t-\log P_{-1},
\qquad
\purple{\check p_t:\ \text{price-level targeting.}}
\]

While under discretion, we have purely static:
\(\hat x_t=-\dfrac{\kappa}{\alpha_x}\hat\pi_t\).

\end{align*}
Let
\[
a=\frac{\alpha_x}{(1+\beta)\alpha_x+\kappa^2}.
\]
Then
\[
\check p_t=a\check p_{t-1}+a\beta E_t\{\check p_{t+1}\}+au_t.
\qquad \forall t

Assume

\qquad \text{undetermined coefficients.}\\

+a u_t,\\

&+

=0.
\end{align*}
\[
\left\{

\end{aligned}
\right.
\Rightarrow
\left\{
\begin{aligned}

\eta&=\frac{\delta}{1-\delta\beta\rho_u}.
\end{aligned}
\right.
\]
Thus
\[
\check p_t=\delta\check p_{t-1}+\frac{\delta}{1-\delta\beta\rho_u}u_t,

\tag{**}

\]

\section{Distorted steady state}

Permanent supply-side distortion:
\[
\bar y^{\,n}<\bar y^{\,e}.
\]
Real imperfections generate a permanent gap between natural and efficient levels of output:
\[
-\frac{U_N}{U_C}=MPN(1-\phi),
\qquad
\phi=1-\frac{1}{\mu}.
\]
Output is inefficiently low.\\[2em]

The welfare losses:
\[
E_0\sum_{t=0}^{\infty}\beta^t
\left[\frac12(\hat\pi_t^2+\alpha_x\hat x_t^2)-\Lambda\hat x_t\right],
\qquad
\text{where }\Lambda=\phi\frac{\lambda}{\varepsilon}>0.
\]
Deviation of the welfare-relevant output gap:
\[
\hat x_t
=
\log X_t-\log \bar X
=
(\log Y_t-\log Y_t^{e})
-
(\log \bar Y-\log \bar Y^{e})
=
(\log Y_t-\log Y^{e})
-
(\log \bar Y^{n}-\log \bar Y^{e}).
\]
\[
\log \bar X
=
\log \bar Y-\log \bar Y^{e}
=
\log \bar Y^{n}-\log \bar Y^{e}
<0.
\]
NKPC constraint:
\[
\hat\pi_t=\beta E_t\{\hat\pi_{t+1}\}+\kappa\hat x_t+u_t,
\qquad
\text{where }u_t=\kappa(\hat y_t^{\,e}-\hat y_t^{\,n})
\text{ is cost-push shock.}
\]

\subsection{Optimal discretionary policy}

\[
\min_{\hat x_t,\hat\pi_t}
\frac12(\hat\pi_t^2+\alpha_x\hat x_t^2)-\Lambda\hat x_t
\qquad
\text{s.t.}\quad
\hat\pi_t=\kappa\hat x_t+v_t,
\quad v_t=\beta E_t\{\hat\pi_{t+1}\}+u_t.
\]
F.O.C.:
\[
\hat x_t
=
\frac{\Lambda}{\alpha_x}
-
\frac{\kappa}{\alpha_x}\hat\pi_t.
\]

\noindent

=\frac{\alpha_x\beta}{\alpha_x+\kappa^2}E_t\{\hat\pi_{t+1}\}

\subsection{Optimal policy under commitment}

\[
\mathcal L

\left[\frac12(\hat\pi_t^2+\alpha_x\hat x_t^2)-\Lambda\hat x_t
+\gamma_t(\hat\pi_t-

F.O.C.:

&\Rightarrow
\hat\pi_t+\gamma_t-\textcolor{red}{\gamma_{t-1}}=0,
\tag{1}\\
\frac{\partial\mathcal L}{\partial \hat x_t}=0
&\Rightarrow
\alpha_x\hat x_t-\kappa\gamma_t-\Lambda=0,
\tag{2}
\end{align*}
\[
\gamma_{-1}=0\qquad
\]
\[
(1)\Rightarrow
\gamma_t
=
\frac{\alpha_x\hat x_t-\Lambda}{\kappa},
\qquad
\gamma_{t-1}
=

\quad
\text{plug into }(2)
\Rightarrow
\hat x_t
=
\hat x_{t-1}
-
\frac{\kappa}{\alpha_x}\hat\pi_t.

Since $\hat\pi_t=\check p_t-\check p_{t-1}$, by iterating forward:

\purple{Zero average inflation, but positive upward bias above $P_{-1}$.}

\begin{enumerate}

  \item more subdued effects on inflation;
  \item commitment avoids the inflation bias under discretionary policy.
