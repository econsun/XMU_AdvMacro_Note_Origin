Business cycle is fluctuation / deviation around trend. Log linearization: take logs to linearize the system and then take deviation from the trend / steady state.

\textcolor{red}{* Can not work with discontinuous functions, eg: zero lower bound, default, work choice,etc.}

\section{Taylor expansions}

Taylor expansion of order $k$ of function $f$ around $a$ is to derive the best approximation of $f$ around $a$ by a polynomial of order $k$.

Suppose $f$ is $n+1$ times differentiable in an interval \textcolor{red}{around $a$}
\[
x\in\left(a-\varepsilon,\,a+\varepsilon\right),
\]
then
\[
f\left(x\right)=f\left(a\right)+\sum_{i=1}^{n}\frac{f^{\left(i\right)}\left(a\right)}{i!}\left(x-a\right)^i
+\underbrace{o\left(\left(x-a\right)^n\right)}_{\text{Peano remainder terms.}}.
\]


eg:

\[
\text{1st order:}\qquad
f\left(x\right)\approx f\left(a\right)+\frac{f'\left(a\right)}{1!}\left(x-a\right)^1
= f\left(a\right)+f'\left(a\right)\left(x-a\right)
\]

\[
\text{2nd order:}\qquad
f\left(x\right)\approx f\left(a\right)+\frac{f'\left(a\right)}{1!}\left(x-a\right)^1+\frac{f''\left(a\right)}{2!}\left(x-a\right)^2
= f\left(a\right)+f'\left(a\right)\left(x-a\right)+\frac{f''\left(a\right)}{2}\left(x-a\right)^2
\]

\section{Taylor expansion on two dimensions}

Suppose $f$ is second-order continuous $\left(C^2\right)$ in a circle around $\left(x_0,y_0\right)$ that contains $\left(x_0+h,y_0+k\right)$.

\textcolor{red}{For a small move $\left(h,k\right)$ around $\left(x_0,y_0\right)$, how to approximate changes in $f$.}

\[
f\left(x_0+h,y_0+k\right)
=
f\left(x_0,y_0\right)
+
\textcolor{red}{\left[f_1\left(x_0,y_0\right)h+f_2\left(x_0,y_0\right)k\right]}
+
\frac{1}{2}
\textcolor{red}{\left[f_{11}\left(x,y\right)h^2+2f_{12}\left(x,y\right)hk+f_{22}\left(x,y\right)k^2\right]}
\]

where
\[
\left(x,y\right)=\left(x_0+ch,y_0+ck\right)\qquad \text{for some } c\in\left(0,1\right).
\]

\textcolor{red}{A vector format:}
\[
\textcolor{red}{
f\left(z_0+\Delta\right)=f\left(z_0\right)+\nabla f\left(z_0\right)^T\Delta+\frac{1}{2}\Delta^T H_f\left(z_0+c\Delta\right)\Delta
}
\]
\[
\textcolor{red}{
\text{where } z_0=\left(x_0,y_0\right),\qquad \Delta=\left(h,k\right)
}
\]

\section{Log linearization}
\[
\textcolor{red}{\text{percentage deviation:}}\qquad
\hat x_t
=
\log X_t
-
\underbrace{\log \bar X}_{\textcolor{red}{Steady state}}
\]

Since $\ Y_t \to 0$ when $\ Y_t$ is sufficiently small, we have
\[
e^{\ Y_t}=e^0+e^0\left(\ Y_t-0\right)=1+\ Y_t
\qquad \Rightarrow \qquad
\ Y_t\approx \log\left(1+\ Y_t\right).
\]

\[
\hat x_t = \log\frac{X_t}{\bar X}
= \log\left(1+\frac{X_t-\bar X}{\bar X}\right)
\approx
\frac{X_t-\bar X}{\bar X}.
\]

\[
\hat x_t \times 100\%
\quad \text{is approximately the percentage deviation w.r.t. S.S.}
\]

\newpage

\[
X_t=\bar X\left(\frac{X_t}{\bar X}\right)=\bar X e^{\log\left(X_t/\bar X\right)}=\bar X e^{\hat x_t}
\]

Taking first-order Taylor approximation:
\[
\bar X e^{\hat x_t}
\approx
\bar X e^0+\bar X e^0\left(\hat x_t-0\right)
=
\bar X+\bar X\hat x_t
=
\left(1+\hat x_t\right)\bar X
\]

\[
\Rightarrow\qquad
X_t=
\overbrace{\bar X}^{\textcolor{red}{\text{Steady state}}}
+
\overbrace{\bar X\hat x_t}^{\textcolor{red}{\text{deviation from steady state}}}
\]

\section{Examples}

\subsection{Linear equation}
\[
Y_t=aX_t
\]

\[
X_t\approx \bar X\left(1+\hat x_t\right)
\qquad \Rightarrow \qquad
Y_t\approx a\bar X\left(1+\hat x_t\right)=a\bar X+a\bar X\hat x_t
\]

Since
\[
Y_t=aX_t,\ \forall t
\qquad \Rightarrow \qquad
\bar Y=a\bar X,
\]
we get
\[
Y_t\approx \bar Y+\bar Y \hat x_t
\qquad \Rightarrow \qquad
Y_t\approx \bar Y+\bar Y \hat y_t
\]
hence
\[
\hat x_t=\hat y_t.
\]

\subsection{Product}
\[
Z_t=X_tY_t
\]

\[
X_t\approx \bar X\left(1+\hat x_t\right), \qquad Y_t\approx \bar Y\left(1+\hat y_t\right)
\]

\[
\Rightarrow\quad
Z_t\approx \bar X\bar Y\left(1+\hat x_t\right)\left(1+\hat y_t\right)
=
\bar X\bar Y\left(1+\hat x_t+\hat y_t+\underbrace{\textcolor{red}{\hat x_t\hat y_t}}_{\textcolor{red}{\approx 0}}\right)
\]

So
\[
Z_t\approx \bar X\bar Y\left(1+\hat x_t+\hat y_t\right).
\]

Since
\[
Z_t=X_tY_t,\ \forall t
\qquad \Rightarrow \qquad
\bar Z=\bar X\bar Y,
\]
we get
\[
Z_t\approx \bar Z+\bar Z\left(\hat x_t+\hat y_t\right)
\qquad \Rightarrow \qquad
Z_t\approx \bar Z+\bar Z\hat z_t
\]
hence
\[
\hat z_t=\hat x_t+\hat y_t.
\]

\subsection{Constant product}
Log linearize equation
\[
X_tY_t=C
\qquad \Rightarrow \qquad
\bar X\bar Y=C.
\]

By (2),
\[
X_tY_t\approx \bar X\bar Y\left(1+\hat x_t+\hat y_t\right)
\qquad \Rightarrow \qquad
C\approx C\left(1+\hat x_t+\hat y_t\right)
\]
so
\[
\hat x_t+\hat y_t=0.
\]

\subsection{Ratio}
\[
\frac{X_t}{Y_t}=C
\qquad \Rightarrow \qquad
\frac{\bar X}{\bar Y}=C.
\]

\[
\frac{X_t}{Y_t}
\approx
\frac{\bar X\left(1+\hat x_t\right)}{\bar Y\left(1+\hat y_t\right)}
\]

Use
\[
\left(1+\hat y_t\right)\left(1-\hat y_t\right)=1+\hat y_t-\hat y_t-\hat y_t\hat y_t
\qquad \Rightarrow \qquad
\frac{1}{1+\hat y_t}\approx \left(1-\hat y_t\right).
\]

Therefore
\[
\frac{X_t}{Y_t}
=
\frac{\bar X}{\bar Y}\left(1+\hat x_t\right)\left(1-\hat y_t\right)
\approx
\frac{\bar X}{\bar Y}\left(1+\hat x_t-\hat y_t-\textcolor{red}{\hat x_t\hat y_t}\right)
\]


So
\[
\frac{X_t}{Y_t}\approx \frac{\bar X}{\bar Y}\left(1+\hat x_t-\hat y_t\right).
\]

Since
\[
\frac{X_t}{Y_t}=\frac{\bar X}{\bar Y}=C,
\]
we have
\[
C=C\left(1+\hat x_t-\hat y_t\right)
\qquad \Rightarrow \qquad
\hat x_t-\hat y_t=0.
\]

\subsection{Sum}
\[
X_t+Y_t=C
\qquad \Rightarrow \qquad
\bar X+\bar Y=C.
\]

\[
X_t+Y_t\approx \bar X\left(1+\hat x_t\right)+\bar Y\left(1+\hat y_t\right)
=\bar X+\bar Y+\bar X\hat x_t+\bar Y\hat y_t
\]
hence
\[
\bar X\hat x_t+\bar Y\hat y_t=0.
\]

\subsection{Budget constraint}
\[
C_t+K_t=Y_t+\left(1-\delta\right)K_{t-1}
\qquad \Rightarrow \qquad
\bar C+\bar K=\bar Y+\left(1-\delta\right)\bar K
\]

\[
C_t\approx \bar C\left(1+\hat c_t\right),\qquad
K_t\approx \bar K\left(1+\hat k_t\right),\qquad
Y_t\approx \bar Y\left(1+\hat y_t\right),\qquad
K_{t-1}\approx \bar K\left(1+\hat k_{t-1}\right)
\]

\[
\Rightarrow\quad
\bar C\left(1+\hat c_t\right)+\bar K\left(1+\hat k_t\right)
=
\bar Y\left(1+\hat y_t\right)+\left(1-\delta\right)\bar K\left(1+\hat k_{t-1}\right)
\]

\[
\Rightarrow\quad
\bar C+\bar C\hat c_t+\bar K+\bar K\hat k_t
=
\bar Y+\bar Y\hat y_t+\left(1-\delta\right)\bar K+\left(1-\delta\right)\bar K\hat k_{t-1}
\]

Using
\[
\bar C+\bar K=\bar Y+\left(1-\delta\right)\bar K,
\]
we obtain
\[
\bar C\hat c_t+\bar K\hat k_t
=
\bar Y\hat y_t+\left(1-\delta\right)\bar K\hat k_{t-1}.
\]

\subsection{Utility FOC}
\[
C_t^{-\sigma}=\lambda_t
\qquad \Rightarrow \qquad
\bar C^{-\sigma}=\bar \lambda
\]

Let
\[
C_t=\bar C e^{\hat c_t},\qquad \lambda_t=\bar \lambda e^{\hat \lambda_t}.
\]

Then
\[
C_t^{-\sigma}
=
\bar C^{-\sigma}e^{-\sigma \hat c_t}
=
\bar \lambda e^{\hat \lambda_t}.
\]

Since $\bar C^{-\sigma}=\bar \lambda$,
\[
e^{-\sigma \hat c_t}=e^{\hat \lambda_t}.
\]

Using
\[
e^{-\sigma \hat c_t}\approx 1-\sigma \hat c_t,
\qquad
e^{\hat \lambda_t}\approx 1+\hat \lambda_t,
\]
we get
\[
1-\sigma \hat c_t=1+\hat \lambda_t
\qquad \Rightarrow \qquad
-\sigma \hat c_t=\hat \lambda_t.
\]

\subsection{C-D production function}
\[
Y_t=A_t K_t^\alpha N_t^{1-\alpha}
\qquad \Rightarrow \qquad
\bar Y=\bar A \bar K^\alpha \bar N^{1-\alpha}
\]

\[
\log Y_t=\log A_t+\alpha \log K_t+\left(1-\alpha\right)\log N_t
\]
\[
\log \bar Y=\log \bar A+\alpha \log \bar K+\left(1-\alpha\right)\log \bar N
\]

Subtract:
\[
\log Y_t-\log \bar Y
=
\left(\log A_t-\log \bar A\right)
+\alpha\left(\log K_t-\log \bar K\right)
+\left(1-\alpha\right)\left(\log N_t-\log \bar N\right)
\]
hence
\[
\hat y_t=\hat a_t+\alpha \hat k_t+\left(1-\alpha\right)\hat n_t.
\]


\subsection{GDP identity}
\[
Y_t=C_t+I_t+G_t
\]

From (5) we know
\[
\hat y_t
=
\frac{\bar C}{\bar Y}\hat c_t
+
\frac{\bar I}{\bar Y}\hat i_t
+
\frac{\bar G}{\bar Y}\hat g_t
=
s_c\hat c_t+s_i\hat i_t+s_g\hat g_t.
\]

\textcolor{red}{share of GDP for each component}

\subsection{Capital law of motion}
\[
K_{t+1}=\left(1-\delta\right)K_t+I_t
\qquad \Rightarrow \qquad
\bar K=\left(1-\delta\right)\bar K+\bar I
\]

\[
\Rightarrow\quad
\bar K\left(1+\hat k_{t+1}\right)
=
\left(1-\delta\right)\bar K\left(1+\hat k_t\right)+\bar I\left(1+\hat i_t\right)
\]

\[
\Rightarrow\quad
\bar K+\bar K\hat k_{t+1}
=
\left(1-\delta\right)\bar K+\left(1-\delta\right)\bar K\hat k_t+\bar I+\bar I\hat i_t
\]

Using
\[
\bar K=\left(1-\delta\right)\bar K+\bar I,
\]
we get
\[
\bar K\hat k_{t+1}=\left(1-\delta\right)\bar K\hat k_t+\bar I\hat i_t.
\]

Thus
\[
\hat k_{t+1}
=
\left(1-\delta\right)\hat k_t+\frac{\bar I}{\bar K}\hat i_t.
\]

Since
\[
\bar K=\left(1-\delta\right)\bar K+\bar I
\qquad \Rightarrow \qquad
\frac{\bar I}{\bar K}=\delta,
\]
we obtain
\[
\hat k_{t+1}=\left(1-\delta\right)\hat k_t+\delta \hat i_t.
\]

\subsection{Euler equation}
\[
U'\left(C_t\right)=\beta E_t\left[U'\left(C_{t+1}\right)R_{t+1}\right]
\]

use CRRA utility:
\[
U\left(C_t\right)=\frac{C_t^{1-\sigma}}{1-\sigma}
\qquad \Rightarrow \qquad
U'\left(C_t\right)=C_t^{-\sigma}
\]

So
\[
C_t^{-\sigma}
=
\beta E_t\left[C_{t+1}^{-\sigma}R_{t+1}\right].
\]

At steady state:
\[
\bar C^{-\sigma}
=
\beta \bar C^{-\sigma}\bar R
\qquad \Rightarrow \qquad
1=\beta \bar R.
\]

Let
\[
C_t=\bar C e^{\hat c_t},
\qquad
C_{t+1}=\bar C e^{\hat c_{t+1}},
\qquad
R_{t+1}=\bar R e^{\hat r_{t+1}}.
\]

Then
\[
\bar C^{-\sigma}e^{-\sigma \hat c_t}
=
\beta E_t\left[\bar C^{-\sigma}e^{-\sigma \hat c_{t+1}}\bar R e^{\hat r_{t+1}}\right].
\]

First-order approximation:
$$
\bar C^{-\sigma}\left(1-\sigma \hat c_t\right)
=
\beta E_t\left[\bar C^{-\sigma}\bar R \left(1-\sigma \hat c_{t+1}\right)\left(1+\hat r_{t+1}\right)\right]
$$

$$
=
\beta E_t\left[\bar C^{-\sigma}\bar R
\left(1+\hat r_{t+1}-\sigma \hat c_{t+1}
-\underbrace{\textcolor{red}{\sigma \hat c_{t+1}\hat r_{t+1}}}_{\color{red}\approx 0}
\right)\right]
$$

Using $1=\beta \bar R$:
\[
1-\sigma \hat c_t
=
1+E_t\left(-\sigma \hat c_{t+1}+\hat r_{t+1}\right).
\]

Hence
\[
-\sigma \hat c_t
=
E_t\left(-\sigma \hat c_{t+1}+\hat r_{t+1}\right)
\]
so
\[
\hat c_t
=
E_t\left(\hat c_{t+1}\right)
-\frac{1}{\sigma}E_t\left(\hat r_{t+1}\right).
\]
